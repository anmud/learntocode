### The Solution 1

```js
var html = '';
var red;
var green;
var blue;
var rgbColor;
var codeRun = 1; 

function getRandomColor( ){
return Math.floor(Math.random() * 256 );
}

for( codeRun = 1; codeRun <= 10; codeRun ++ ){
 red = getRandomColor();
 green = getRandomColor();
 blue = getRandomColor();
 rgbColor = 'rgb(' + red + ',' + green + ',' + blue + ')'; 
 html += '<div style="background-color:' + rgbColor + '"></div>';
}

document.write(html);
```
